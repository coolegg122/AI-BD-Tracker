```javascript
const groupHistory = (items) => {
  const groups = {};
  const now = new Date();
  
  items.forEach(item => {
    const date = new Date(item.created_at.replace(' ', 'T'));
    const diffDays = Math.floor((now - date) / (1000 * 60 * 60 * 24));
    
    let groupName = "";
    if (diffDays < 7) {
      groupName = "This Week";
    } else if (diffDays < 14) {
      groupName = "Last Week";
    } else {
      groupName = date.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });
    }
    
    if (!groups[groupName]) groups[groupName] = [];
    groups[groupName].push(item);
  });
  
  return groups;
};
```

```python
@app.get("/api/v1/smart-input/archive", response_model=List[schemas.SmartInputArchiveResponse])
def get_smart_input_archive(
    skip: int = 0,
    limit: int = 100,
    db: Session = Depends(database.get_db),
    current_user: models.User = Depends(get_current_active_user)
):
    """获取当前用户的 Smart Input 历史档案"""
    archives = db.query(models.SmartInputArchive).filter(
        models.SmartInputArchive.user_id == current_user.id
    ).order_by(
        models.SmartInputArchive.created_at.desc()
    ).offset(skip).limit(limit).all()
    
    return archives
```
